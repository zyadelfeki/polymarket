import httpx
from decimal import Decimal

class KalshiClient:
    def __init__(self, api_key):
        self.base_url = "https://trading-api.kalshi.com/trade-api/v2"
        self.api_key = api_key
        self.client = httpx.AsyncClient(
            headers={"Authorization": f"Bearer {api_key}"}
        )
        self.rate_limit = 10  # req/sec
        
    async def get_market_orderbook(self, market_id):
        """Get current YES/NO orderbook"""
        try:
            resp = await self.client.get(
                f"{self.base_url}/markets/{market_id}/orderbook"
            )
            data = resp.json()
            return {
                'bid_yes': Decimal(str(data['bids'][0]['price'])) if data['bids'] else None,
                'ask_yes': Decimal(str(data['asks'][0]['price'])) if data['asks'] else None,
                'volume_yes': int(data['bids'][0]['quantity']) if data['bids'] else 0,
                'bid_no': Decimal("1.00") - Decimal(str(data['asks'][0]['price'])) if data['asks'] else None,
                'ask_no': Decimal("1.00") - Decimal(str(data['bids'][0]['price'])) if data['bids'] else None,
            }
        except Exception as e:
            print(f"Kalshi orderbook error: {e}")
            return None
    
    async def place_order(self, market_id, side, quantity, price):
        """Place order on Kalshi"""
        try:
            payload = {
                'market_id': market_id,
                'side': side.upper(),  # BUY or SELL
                'quantity': int(quantity),
                'price': float(price),
                'order_type': 'LIMIT'
            }
            resp = await self.client.post(
                f"{self.base_url}/orders",
                json=payload
            )
            order = resp.json()
            return {
                'id': order['order_id'],
                'status': 'filled' if order['status'] == 'FILLED' else 'pending',
                'quantity': order['quantity'],
                'price': order['price']
            }
        except Exception as e:
            print(f"Kalshi order error: {e}")
            return {'status': 'failed'}

    async def detect_news_alert(self, tweet_text, author):
        """Parse tweet for actionable news (confidence score)"""
        text_lower = tweet_text.lower()
        confidence = 0.0
        keywords_found = []
        
        for keyword, weight in self.hot_keywords.items():
            if keyword in text_lower:
                confidence += weight * 0.1  # 10% per keyword
                keywords_found.append(keyword)
        
        official_accounts = {'ap', 'reuters', 'whitehouse', 'federalreserve'}
        if author.lower() in official_accounts:
            confidence += 0.3
        
        if re.search(r'\d+(\.\d+)?%|\$\d+', tweet_text):
            confidence += 0.2
        
        return {
            'confidence': min(confidence, 1.0),
            'keywords': keywords_found,
            'timestamp': datetime.now()
        }
    
    async def match_to_market(self, news_text, poly_markets):
        """Find best matching Polymarket"""
        best_match = None
        best_score = 0.0
        
        for market in poly_markets:
            title = market['title'].lower()
            
            news_words = set(news_text.lower().split())
            market_words = set(title.split())
            overlap = len(news_words & market_words)
            
            volume_score = min(market['volume'] / 1000, 1.0)
            
            combined_score = (overlap * 0.6) + (volume_score * 0.4)
            
            if combined_score > best_score:
                best_score = combined_score
                best_match = market
        
        return best_match if best_score > 0.3 else None

from decimal import Decimal
import asyncio

class MarketMaker:
    def __init__(self, capital, kelly_sizer):
        self.capital = capital
        self.kelly_sizer = kelly_sizer
        self.max_inventory = capital * Decimal("0.10")  # 10% per market
        self.markets_selected = []
        
    async def select_markets(self, poly_client, all_markets):
        """Choose high-volume, low-volatility markets"""
        scored_markets = []
        
        for market in all_markets:
            volume_24h = market['volume_24h']
            volatility = market['bid_ask_spread']  # Wider = more volatile
            
            score = (volume_24h / 1000) / (volatility + 0.01)
            
            scored_markets.append({
                'market': market,
                'score': score
            })
        
        self.markets_selected = sorted(
            scored_markets, 
            key=lambda x: x['score'], 
            reverse=True
        )[:5]
        
        return self.markets_selected
    
    async def place_quotes(self, market, poly_client):
        """Post bid/ask quotes"""
        mid_price = Decimal(str(market['mid_price']))
        bid_ask_spread = Decimal("0.04")  # 4% spread
        
        bid = mid_price - (bid_ask_spread / Decimal("2"))
        ask = mid_price + (bid_ask_spread / Decimal("2"))
        
        quote_size = self.max_inventory / Decimal("10")  # 1% per quote
        
        bid_order = await poly_client.place_order(
            market_id=market['id'],
            side='buy',
            quantity=float(quote_size),
            price=float(bid)
        )
        
        ask_order = await poly_client.place_order(
            market_id=market['id'],
            side='sell',
            quantity=float(quote_size),
            price=float(ask)
        )
        
        return [bid_order, ask_order]
    
    async def rebalance_every_10_minutes(self, poly_client):
        """Adjust quotes based on new mid-price"""
        while True:
            for selected in self.markets_selected:
                market = selected['market']
                market_update = await poly_client.get_market(market['id'])
                
                await poly_client.cancel_all_orders(market['id'])
                await self.place_quotes(market_update, poly_client)
            
            await asyncio.sleep(600)  # Every 10 minutes

from decimal import Decimal
import math

class AdaptiveKellySizer:
    def __init__(self, config):
        self.kelly_fraction = Decimal(str(config.get('kelly_fraction', 0.25)))
        self.consecutive_wins = 0
        self.consecutive_losses = 0
        self.total_trades = 0
        self.win_rate = Decimal("0.50")
        
    def calculate_bet_size(self, capital, win_prob, win_payout, loss_payout):
        """
        Calculate optimal bet size using Kelly Criterion
        
        Formula: f* = (bp - q) / b
        where: b = odds-1, p = win prob, q = loss prob
        
        Simplified: f = (2*p - 1) for 1:1 payoff
        """
        if win_prob < Decimal("0.51"):  # Only bet if >51% edge
            return Decimal("0")
        
        bp = Decimal(str(win_payout)) - Decimal("1")
        q = Decimal("1") - win_prob
        p = win_prob
        
        kelly_full = ((bp * p) - q) / bp
        
        kelly_applied = kelly_full * self.kelly_fraction
        
        kelly_applied = max(kelly_applied, Decimal("0.0001"))  # Min bet
        kelly_applied = min(kelly_applied, Decimal("0.10"))    # Max 10% per trade
        
        bet_size = capital * kelly_applied
        
        return bet_size
    
    def record_trade_result(self, won, size, pnl):
        """Track results for adaptive Kelly"""
        self.total_trades += 1
        
        if won:
            self.consecutive_wins += 1
            self.consecutive_losses = 0
        else:
            self.consecutive_losses += 1
            self.consecutive_wins = 0
        
        if self.total_trades > 0:
            self.win_rate = Decimal(str(self.consecutive_wins + self.consecutive_losses)) / Decimal(str(self.total_trades))
    
    def get_stats(self):
        """Return Kelly stats for logging"""
        return {
            'kelly_fraction': float(self.kelly_fraction),
            'consecutive_wins': self.consecutive_wins,
            'consecutive_losses': self.consecutive_losses,
            'total_trades': self.total_trades,
            'win_rate': float(self.win_rate)
        }
